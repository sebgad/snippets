#File/Location file to set up file list to copy
#Format must be Source=Target
# e.g for a single file:
# 	"C:\SourceFile.txt"="C:\TargetFolder\TargetFile.txt"

#	"C:\SourceFolder\*"="C:\TargetFolder1\TargetFolder2"
# Lines start with "#" will be ignored
# ATTENTION: Files will be overwritten on default. Change it in copyFilesInList.ps1 if you do not want it.
